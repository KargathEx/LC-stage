class Solution {
public:
  vector<int> canSeePersonsCount(vector<int>& n) {
    vector<int> res;
    for (int i = n.size() - 1; i >= 0; --i)
    {
      int j = i + 1;
      int cnt = 0;
      int max_ = 0;
      while (j < n.size())
      {
        if (n[j] > max_) {
          ++cnt; max_ = n[j];
        }
        if (n[j] < n[i])++j;
        else break;
      }
  
      res.push_back(cnt);
    }
    reverse(all(res));
    return res;
  }
};
